function submitForm(e) { e.preventDefault(); const form = e.target; const name = form.querySelector('input[type="text"]').value; const phone = form.querySelector('input[type="tel"]').value; const btn = form.querySelector('.btn-primary'); const BOT_TOKEN = 'ВАШ_ТОКЕН_БОТА'; const CHAT_ID = 'ВАШ_ID_ЧАТА'; const text = `🆕 Новая заявка с лендинга!\n👤 Имя: ${name}\n📱 Телефон: ${phone}\n📱 Модель: Xiaomi 17 12/256 Blu`; btn.textContent = '⏳ Отправка...'; btn.disabled = true; fetch(`https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: CHAT_ID, text: text }) }) .then(res => res.json()) .then(data => { if (data.ok) { btn.textContent = '✅ Отправлено!'; btn.style.background = '#22c55e'; form.reset(); } else { throw new Error('Ошибка'); } }) .catch(() => { btn.textContent = '❌ Ошибка'; btn.style.background = '#ef4444'; }) .finally(() => { setTimeout(() => { btn.textContent = '✅ Заказать Xiaomi 17 Blu'; btn.style.background = ''; btn.disabled = false; }, 3000); }); }
